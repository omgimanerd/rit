\section*{Review 4}
\begin{enumerate}
  \item ``Vandermonde Matrix''
  \item Finding the kernel of \( D^n \)
  \begin{align*}
    D^2 &= D\circ D \\
    D^3 &= D\circ D\circ D \\
    D^n &= D\circ D^{n-1}
  \end{align*}
  \item This boils down to solving a differential equation.
  \item \( x^n-1 = (x-1)(\sum_{i=1}^{n-1}x^i) \)
  \[ L^n-II = (L-I)(\sum_{i=0}^{n-1}L^i) \]
  Use this for part (a) and (c).
\end{enumerate}
