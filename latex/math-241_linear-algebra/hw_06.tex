\subsubsection*{Exercise 30}
Prove Theorem 3.4(a)-(c).
\begin{enumerate}[(a)]
  \item \( (A^T)^T = A \)
  \begin{align*}
    B &= A^T \\
    \therefore b_{ij} &= a_{ji} \\
    C &= B^T = (A^T)^T \\
    \therefore c_{ij} &= b_{ji} \\
    \therefore a_{ij} &= b_{ji} = c_{ij} \\
    A &= C = (A^T)^T
  \end{align*}
  \item \( (A+B)^T = A^T+B^T \)
  \begin{align*}
    (A+B)^T &= C \\
    a_{ij}+b_{ij} &= c_{ij} \\
    A^T+B^T &= C \\
    \therefore A^T+B^T &= (A+B)^T
  \end{align*}
  \item \( (kA)^T = k(A^T) \)
  \begin{align*}
    B &= A^T \\
    \therefore b_{ij} &= a_{ji} \\
    kB &= k(A^T) \\
    kb_{ij} &= k(a_{ji}) \\
    &= (kA)^T \\
    k(B^T) &= k(A^T) = (kA)^T
  \end{align*}
\end{enumerate}

    0 & a_{11}b_{12}+a_{12}b_{22}-b_{11}a_{12}+b_{12}a_{22} \\
    a_{21}b_{11}+a_{22}b_{21}-b_{21}a_{11}+b_{22}a_{21} & 0
  \end{bmatrix} \\
  &\ne I_2
\end{align*}
